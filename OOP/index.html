Base Classes and Sample Usage


Base Class: Person
class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    describeRole() {
        throw new Error("Method 'describeRole()' must be implemented.");
    }
}

Extended Class: Teacher
class Teacher extends Person {
    constructor(name, age, subject) {
        super(name, age);
        this.subject = subject;
    }
    describeRole() {
        return `${this.name} is a teacher who teaches ${this.subject}.`;
    }
}

Sample Usage:
const teacher = new Teacher("Alice", 30, "Math");
console.log(teacher.describeRole()); // Alice is a teacher who teaches Math.

Base Class: Food
class Food {
    constructor(type) {
        this.type = type;
    }
    prepare() {
        throw new Error("Method 'prepare()' must be implemented.");
    }
}

Extended Class: FastFood
class FastFood extends Food {
    constructor() {
        super("Fast Food");
    }
    prepare() {
        return "Preparing fast food!";
    }
}

Sample Usage:
const burger = new FastFood();
console.log(burger.prepare()); // Preparing fast food!

Base Class: Animal
class Animal {
    constructor(species) {
        this.species = species;
    }
    makeSound() {
        throw new Error("Method 'makeSound()' must be implemented.");
    }
}

Extended Class: Dog
class Dog extends Animal {
    constructor(name) {
        super("Dog");
        this.name = name;
    }
    makeSound() {
        return `${this.name} says Woof!`;
    }
}

Sample Usage:
const dog = new Dog("Buddy");
console.log(dog.makeSound()); // Buddy says Woof!
